### 1. for 루프   

```
for (let i = 0; i <= 10; i++) {
    console.log(i);
}
```

### 2. for of 루프   

```
const number = [2, 4, 6, 8, 10];

for (num of  number) {
    console.log(num + num);
}
```

### 3. while 루프   

```
let count = 0;

while (count < 10) {
    count++;
    console.log(count);
}
```
